# 🧱 3. Estrutura da Linguagem

## 3.1 — Bloco de Robô
$ isaac show Ref_AC
# Output:
## <Ref_AC>
##   Descarte qualquer AC existente;
##   Registre e tome como referência o Arquivo de Conteúdo (AC);
##   Use este AC em substituição a qualquer outro;
##   Analise o AC;
## </Ref_AC>

## 3.2 — Chamada de Execução
$ isaac run Ref_AC --AC="Produção Literária Acadêmica Com IA.pdf"
$ isaac run Ref_AD --AD="Diretrizes - Produção Literária Acadêmica Com IA (AD).pdf"
$ isaac run Script_Fichamento --x=91 --RBx="NBR-10520-CITAÇÕES.pdf"

## 3.3 — Comandos Internos
$ isaac run Ref_AF --AF="Fichamentos - Produção Literária Acadêmica Com IA.pdf"
$ isaac run Ref_RB --x=74 --RBx="abs-guide.pdf"

# ⚙️ 4. Arquitetura de Execução
$ isaac parse Ref_AC
$ isaac interpret Ref_AC

# 🔧 5. Parametrização Externa
$ isaac run Ref_RB --x=74 --RBx="abs-guide.pdf"

# 🔁 6. Persistência de Variáveis
$ isaac run Ref_AC
$ isaac run Ref_AD
$ isaac run Script_Fichamento

# 🧠 7. Reconhecimento de Entidades Externas
$ isaac run Replicar_EGO
$ isaac run Glossário

# 📦 8. Escalabilidade Funcional
$ isaac run Fichamento_Rápido --x=74 --RBx="abs-guide.pdf"
$ isaac batch --scripts="Ref_AC, Ref_AD, Script_Fichamento"

# 🔄 9. Execução por Intervalo (Range)
$ isaac loop Análise_Editorial_Seção --SAA=2
## Interpretação:
### → isaac run Análise_Editorial_Seção --SAA=2
